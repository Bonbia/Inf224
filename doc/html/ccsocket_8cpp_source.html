84int Socket::setAddress(SOCKADDR_IN& addr, const string& host, int port) {
85#if defined(_WIN32) || defined(_WIN64)
86 addr = {};
87 addr.sin_family = AF_INET;
88 addr.sin_port = htons(port);
89 inet_pton(AF_INET, host.data(), &addr.sin_addr.s_addr);
90#else
91 addr = {};
92 struct hostent* hent = NULL;
93 // gethostbyname() is obsolete!
94 if (host.empty() || !(hent = ::gethostbyname(host.c_str()))) return -1; // host not found
95 addr.sin_family = AF_INET;
96 addr.sin_port = htons(port);
97 // NB: data might be misaligned but correct result because memcpy() is used
98 ::memcpy(&addr.sin_addr, hent->h_addr_list[0], hent->h_length);
99#endif
100 return 0;
101}

211int ServerSocket::bind(int port, int backlog) {
212 if (sockfd_ == INVALID_SOCKET) return Socket::InvalidSocket;
213
214 SOCKADDR_IN addr{};
215 addr.sin_family = AF_INET;
216 addr.sin_port = htons(port);
217 addr.sin_addr.s_addr = INADDR_ANY;
218
219 if (::bind(sockfd_, (SOCKADDR*)&addr, sizeof(addr)) < 0) return -1;
220
221#if defined(_WIN32) || defined(_WIN64)
222#else
223 socklen_t taille = sizeof addr;
224 if (::getsockname(sockfd_, (SOCKADDR*)&addr, &taille) < 0) return -1;
225#endif
226
227 // le serveur se met en attente sur le socket d'ecoute
228 // listen s'applique seulement aux sockets de type SOCK_STREAM ou SOCK_SEQPACKET.
229 if (::listen(sockfd_, backlog) < 0) return -1;
230 return 0;
231}

342SOCKSIZE SocketBuffer::readLine(string& str) {
343 str.clear();
344 if (!sock_) return Socket::InvalidSocket;
345 if (!in_) in_ = new InputBuffer(insize_);
346
347 while (true) {
348 if (retrieveLine(str, in_->remaining)) return str.length() + 1;
349 // - received > 0: data received
350 // - received = 0: nothing received (shutdown or empty message)
351 // - received < 0: an error occurred
352 SOCKSIZE received = sock_->receive(in_->begin, in_->end - in_->begin);
353 if (received <= 0) return received; // -1 (error) or 0 (shutdown)
354 if (retrieveLine(str, received)) return str.length() + 1;
355 }
356}

359bool SocketBuffer::retrieveLine(string& str, SOCKSIZE received) {
360 if (received <= 0 || in_->begin > in_->end) {
361 in_->begin = in_->buffer;
362 return false;
363 }
364
365 // search for separator
366 char* sep = nullptr;
367 int sepLen = 1;
368
369 if (insep_ < 0) { // means: '\r' or '\n' or "\r\n"
370 for (char* p = in_->begin; p < in_->begin + received; ++p) {
371 if (*p == '\n') {
372 sep = p;
373 sepLen = 1;
374 break;
375 }
376 else if (*p == '\r') {
377 sep = p;
378 if (p < in_->begin + received - 1 && *(p + 1) == '\n') {
379 sepLen = 2;
380 }
381 break;
382 }
383 }
384 }
385 else {
386 for (char* p = in_->begin; p < in_->begin + received; ++p)
387 if (*p == insep_) {
388 sep = p;
389 break;
390 }
391 }
392
393 if (sep) {
394 str.append(in_->begin, sep - in_->begin);
395 in_->remaining = received - (sep + sepLen - in_->begin);
396 in_->begin = sep + sepLen;
397 return true;
398 }
399 else {
400 str.append(in_->begin, received);
401 in_->begin = in_->buffer;
402 in_->remaining = 0;
403 return false;
404 }
405}

408SOCKSIZE SocketBuffer::writeLine(const string& str) {
409 if (!sock_) return Socket::InvalidSocket;
410
411 size_t strlen = str.length();
412 // a negature value of outsep means that \r\n must be added
413 size_t msglen = strlen + (outsep_ < 0 ? 2 : 1);
414
415 // if msglen is not too large, try to write string + outsep in one block
416 if (msglen <= outsize_) {
417 char* buf = new char[msglen];
418 ::memcpy(buf, str.data(), strlen);
419
420 if (outsep_ >= 0) buf[msglen - 1] = char(outsep_);
421 else {
422 buf[msglen - 2] = '\r';
423 buf[msglen - 1] = '\n';
424 }
425 auto stat = write(buf, msglen);
426 delete[] buf;
427 return stat;
428 }
429 else {
430 SOCKSIZE sent = write(str.data(), strlen);
431 char buf[] = {char(outsep_), 0};
432 if (outsep_ >= 0) sent += sock_->send(buf, 1);
433 else sent += sock_->send("\r\n", 2);
434 return sent;
435 }
436}

439SOCKSIZE SocketBuffer::write(const char* s, size_t len) {
440 if (!sock_) return Socket::InvalidSocket;
441 const char* begin = s;
442 const char* end = s + len;
443 SOCKSIZE totalSent = 0;
444
445 while (begin < end) {
446 // - sent > 0: data sent
447 // - sent = 0: file was shutdown
448 // - sent < 0: an error occurred
449 SOCKSIZE sent = sock_->send(begin, end - begin);
450 if (sent <= 0) return sent; // -1 (error) or 0 (shutdown)
451 begin += sent;
452 totalSent += sent;
453 }
454 return totalSent;
455}
