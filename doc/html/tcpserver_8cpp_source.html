44// infinite loop that processes incoming requests on a TCPServer::Cnx connection.
45void SocketCnx::processRequests() {
46 while (true) {
47 std::string request, response;
48
49 // read the incoming request sent by the client
50 // SocketBuffer::readLine() lit jusqu'au premier délimiteur (qui est supprimé)
51 auto received = sockbuf_->readLine(request);
52
53 if (received < 0) {
54 server_.error("Read error");
55 break;
56 }
57
58 if (received == 0) {
59 server_.error("Connection closed by client");
60 break;
61 }
62
63 // processes the request
64 if (!server_.callback_) {
65 response = "OK";
66 }
67 // closes the connection with this client if the callback returns false
68 else if (!server_.callback_(request, response)) {
69 server_.error("Closing connection with client");
70 break;
71 }
72
73 // a response is always sent to the client (otherwise it might block)
74 // writeLine() response folled by a \n delimiter
75 auto sent = sockbuf_->writeLine(response);
76
77 if (sent < 0) {
78 server_.error("Write error");
79 break;
80 }
81
82 if (sent == 0) {
83 server_.error("Connection closed by client");
84 break;
85 }
86 }
87
88 // free resources and kills thread
89 delete this;
90}

101int TCPServer::run(int port) {
102 int status = servsock_.bind(port); // lier le ServerSocket a ce port
103
104 if (status < 0) {
105 error("Can't bind on port: " + to_string(port));
106 return status; // returns negative value, see Socket::bind()
107 }
108
109 while (true) {
110 if (auto* socket = servsock_.accept()) {
111 // lance la lecture des messages de ce socket dans un thread
112 new SocketCnx(*this, socket);
113 }
114 else error("input connection failed");
115 }
116 return 0; // means OK
117}
